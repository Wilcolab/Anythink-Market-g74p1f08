import re

def to_snake_case(text):
    # Replace spaces and hyphens with underscores
    text = re.sub(r'[\s-]+', '_', text)
    # Add underscore before capital letters (not at the start)
    text = re.sub(r'(?<!^)(?=[A-Z])', '_', text)
    # Lowercase and remove any non-alphanumeric/underscore
    text = re.sub(r'[^a-zA-Z0-9_]', '', text)
    return text.lower()

# Example usage:
print(to_snake_case('ConvertThis Text-to_snakeCase!'))  # Output: convert_this_text_to_snake_case
